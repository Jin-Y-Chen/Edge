Setup — save capture script
cat > ~/capture_arducam.py << 'EOF'
import numpy as np, cv2, subprocess, sys

def capture_frame(device='/dev/video0', raw='/tmp/arducam.raw'):
    subprocess.run(['v4l2-ctl', '-d', device,
        '--stream-mmap', '--stream-count=1',
        f'--stream-to={raw}'], capture_output=True)
    frame = np.fromfile(raw, dtype=np.uint16).reshape(1200, 1920)
    black, white = 1729, 4740
    scaled = (np.clip(frame.astype('int32') - black, 0,
              white - black) * 255 // (white - black)).astype('uint8')
    return cv2.cvtColor(scaled, cv2.COLOR_BayerGR2BGR)

if __name__ == '__main__':
    out = sys.argv[1] if len(sys.argv) > 1 else '/tmp/arducam_output.png'
    cv2.imwrite(out, capture_frame())
    print(f'Saved: {out}')
EOF
